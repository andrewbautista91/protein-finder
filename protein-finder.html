function handleSearch(query) {

    const dropdown = document.getElementById('searchDropdown');

    if (!query || query.length < 1) {

        dropdown.classList.remove('open');

        return;

    }


    const q = query.toLowerCase();

    const matches = Object.entries(restaurantData).filter(([key, r]) =>

        r.name.toLowerCase().includes(q) ||

        r.category.toLowerCase().includes(q)

    );


    if (matches.length === 0) {

        dropdown.innerHTML = `<div class="no-results">No restaurants found for "${query}"<br><small>Try the 📸 Photo tab to scan any menu!</small></div>`;

    } else {

        dropdown.innerHTML = matches.map(([key, r]) => `

            <div class="dropdown-item" onclick="selectRestaurant('${key}')">

                <div class="dropdown-emoji">${r.emoji}</div>

                <div class="dropdown-info">

                    <div class="dropdown-name">${r.name}</div>

                    <div class="dropdown-category">${r.category}</div>

                </div>

                <div class="dropdown-protein">Up to ${r.topProtein}g</div>

            </div>

        `).join('');

    }


    dropdown.classList.add('open');

}

    document.getElementById('searchDropdown').classList.remove('open');

    document.getElementById('searchInput').value = restaurantData[key].name;

}

function parseMenuText(text) {

    const lines = text.split(/[\n,;|•·]+/).map(l => l.trim()).filter(l => l.length > 3);

    const found = [];

    lines.forEach(line => {

        const lower = line.toLowerCase();

        let bestMatch = null, bestLen = 0;

        for (const [kw, data] of Object.entries(proteinKeywords)) {

            if (lower.includes(kw) && kw.length > bestLen) {

                bestMatch = { kw, ...data };

                bestLen = kw.length;

            }

        }

        if (bestMatch) {

            let { protein, cal, tag } = bestMatch;

            if (boostWords.some(w => lower.includes(w)))  { protein = Math.round(protein*1.4); cal = Math.round(cal*1.3); }

            if (reduceWords.some(w => lower.includes(w))) { protein = Math.round(protein*0.6); cal = Math.round(cal*0.7); }

            const priceMatch = line.match(/\$[\d.]+/);

            const price = priceMatch ? priceMatch[0] : "—";

            const name = line.replace(/\$[\d.]+/,'').replace(/^\d+\.?\s*/,'').trim().slice(0,55);

            if (name.length > 2) found.push({ name, protein, calories:cal, price, category:"Menu Item", diet:tag, desc:`Estimated — contains ${bestMatch.kw}` });

        }

    });

    const unique = [], seen = new Set();

    found.forEach(item => {

        const key = item.name.toLowerCase().slice(0,20);

        if (!seen.has(key)) { seen.add(key); unique.push(item); }

    });

    return unique.sort((a,b) => b.protein - a.protein).slice(0,10);

}

function setFilter(type, btn) {

    currentFilter = type;

    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));

    btn.classList.add('active');

    if (currentItems.length) renderItems(currentItems);

}


function applyFilter(items) {

    switch (currentFilter) {

        case 'high':  return items.filter(i => i.protein >= 30);

        case 'lean':  return items.filter(i => i.calories / i.protein < 15);

        case 'value': return [...items].sort((a,b) => {

            const pA = parseFloat(a.price.replace('$',''))||99;

            const pB = parseFloat(b.price.replace('$',''))||99;

            return (pA/a.protein)-(pB/b.protein);

        });

        default: return [...items].sort((a,b) => b.protein - a.protein);

    }

}

function showResults(data) {

    document.getElementById('loading').style.display  = 'none';

    document.getElementById('errorBox').style.display = 'none';

    document.getElementById('results').style.display  = 'block';


    currentItems = data.items;

    maxProtein   = Math.max(...data.items.map(i => i.protein));


    document.getElementById('restaurantName').textContent        = data.name;

    document.getElementById('tipBox').textContent                = data.tip;

    document.getElementById('photoDisclaimer').style.display     = data.isPhoto ? 'block' : 'none';


    const avgProt = Math.round(data.items.reduce((s,i) => s+i.protein, 0) / data.items.length);

    const topItem = [...data.items].sort((a,b) => b.protein - a.protein)[0];

    const leanCnt = data.items.filter(i => i.calories / i.protein < 15).length;


    document.getElementById('summaryBar').innerHTML = `

        <div class="summary-stat"><div class="value">${topItem.protein}g</div><div class="label">Top Item</div></div>

        <div class="summary-stat"><div class="value">${avgProt}g</div><div class="label">Avg Protein</div></div>

        <div class="summary-stat"><div class="value">${leanCnt}</div><div class="label">Lean Options</div></div>`;


    renderItems(data.items);

    setTimeout(() => document.getElementById('results').scrollIntoView({ behavior:'smooth', block:'start' }), 100);

}


function renderItems(items) {

    const filtered   = applyFilter(items);

    const rankColors = ['gold','silver','bronze'];

    const rankEmojis = ['🥇','🥈','🥉'];


    document.getElementById('itemsList').innerHTML = filtered.map((item, idx) => {

        const pct       = Math.round((item.protein / maxProtein) * 100);

        const rankClass = rankColors[idx] || 'other';

        const rankLabel = rankEmojis[idx] || `#${idx+1}`;

        const pPerCal   = (item.protein / item.calories * 100).toFixed(1);

        return `

        <div class="protein-card">

            <div class="rank ${rankClass}">${rankLabel}</div>

            <div class="item-info">

                <div class="item-name">${item.name}</div>

                <div class="item-desc">${item.desc}</div>

                <div class="item-tags">

                </div>

            </div>

                <div class="protein-amount">${item.protein}g</div>

                <div class="protein-label">Protein</div>

            </div>

        </div>`;

    }

}
